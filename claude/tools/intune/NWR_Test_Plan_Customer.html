Quick Validation Script

Run on test device as Administrator:

# Quick validation
$Tests = @()

# Wi-Fi
$WiFi = Get-NetAdapter | Where-Object { $_.InterfaceDescription -match 'Wi-Fi' }
$Tests += @{ Test = "Wi-Fi Disabled"; Pass = ($WiFi.Status -eq 'Disabled') }

# Firewall
$FW = (Get-NetFirewallProfile | Where-Object { $_.DefaultOutboundAction -eq 'Block' }).Count -eq 3
$Tests += @{ Test = "Firewall Blocked"; Pass = $FW }

# Connectivity
$Bastion = Test-NetConnection portal.azure.com -Port 443 -InformationLevel Quiet
$Tests += @{ Test = "Azure Bastion"; Pass = $Bastion }

$Google = Test-NetConnection www.google.com -Port 443 -InformationLevel Quiet
$Tests += @{ Test = "Google Blocked"; Pass = (-not $Google) }

# BitLocker
$BL = (Get-BitLockerVolume -MountPoint C:).ProtectionStatus -eq 'On'
$Tests += @{ Test = "BitLocker On"; Pass = $BL }

# Results
$Tests | ForEach-Object {
    Write-Host "$($_.Test): $(if ($_.Pass) { 'PASS' } else { 'FAIL' })"
}
